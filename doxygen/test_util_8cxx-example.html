test_util.cxx

This example shows how to use the classes of the Utilities module

#include "util/util.hxx"
#include <sstream>
void check (bool expr) {
if (!expr) {
throw std::string ("Error");
}
}
void testTimer () {
Timer timer;
// do something
std::cout << "Testing Timer..." << std::endl;
std::cout << "Elapsed time: " << timer.get() << " s." << std::endl;
}
void testString () {
std::cout << "Testing String..." << std::endl;
// Usage example
String s ("foobar");
check (s.startsWith ("foo"));
// Additional tests
check (s.startsWith ("bar") == false);
}
void testTee () {
std::cout << "Testing Tee" << std::endl;
std::stringstream stream;
Tee tee (std::cout, stream);
tee << "foo" << std::endl;
std::cout << stream.str() << std::endl;
}
int main () {
try {
testTimer();
testString();
testTee();
}
catch (...) {
std::cerr << "Caught exception!" << std::endl;
return 1;
}
return 0;
}
